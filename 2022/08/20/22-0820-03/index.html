汇编Day3

[bx]、loop指令

[bx]

前面day2天学习中，我们了解到了[0]的用法，如下：

mov bx,1000H 
mov ds,bx	
mov al,[0]

但这里中括号中的0是常量，我们还可放变量，比如：[bx]

;如果ds已经是2000H
mov bx,10
mov ax,[bx]		;这个指令就相当于mov ax,2000H:10,也就是把2000H:10这个字型数据给ax
;如果是mov dl,[0]，那么这个是把字节型数据给mov，因为dl只有8位

inx bx 指令的作用是让bx+1，相当于bx=bx+1或者是bx++

loop指令

loop 指令，在汇编里面代表循环，类似于c语言的while、for循环。

loop 指令的使用格式：loop 标号，CPU 执行 loop 指令时，首先执行 CX = CX-1，然后判断 CX 的值，CX的值如果不为0，就会跳到标号处执行程序，如果为0，就按顺序向下执行。

;比如：如果要用指令实现2的三次方，可用如下方法
mov ax,2	;ax=2
add ax,ax	;ax=2+2=4  
add ax,ax	;ax=4+4=8
;但是这里如果要实现2的12次方呢？？？用上面的方法一直加下去就不太好，这里就用循环实现，用如下代码实现
assume cs:code
code segment    
	mov ax,2    
	mov cx,11    
	
s:  add ax,ax    
	loop s            
	
	mov ax,4c00h    
	int 21h
code ends
end

对上面2的12次方debug下，-u后我们发现ayu.exe会先执行mov ax,0002，然后再执行mov cx,000B，一直向下执行下去，然后到loop指令的时候，会再跳回到上面的add ax,ax指令

然后我们可以-t来一条条的指令看下

注意，上面的第二条指令mov cx,000B，这里000B实际就是十进制的11，也就是源程序中的mov cx,11，当然mov cx,11也可以写成mov cx,0BH，这里必须是0BH，不能写成BH，在汇编中数据不能以字母开头，所以在前面加个0

cx存放的是程序（机器码）长度（字节数），一共15个字节，也就是十六进制的F

作业:
编程，用加法计算123*236,结果存在ax中
123*236就是236个123相乘，即123×123×123×123...
代码如下：
code segment
	mov ax,123
	mov cx,236
s:	add ax,ax
	loop s
	mov ax,4c00h
	int 21h
code ends
end

Debug和汇编编译器处理指令的区别

用debug输入如下指令

mov ax,2000h
mov ds,ax
mov al,[0]
mov bl,[1]
mov cl,[2]
mov dl,[3]

然后用编译器编译和debug如下程序

assume cs:code
code segment
	mov ax,2000h
	mov ds,ax
	mov al,[0]
	mov bl,[1]
	mov cl,[2]
	mov dl,[3]
	
	mov ax,4c00h
	int 21h
code ends
end

那么，我们如何解决上面的问题呢？？把代码改成如下，添加上段前缀ds:，然后编译器就会把[0]当成地址处理了

assume cs:code
code segment
	mov ax,2000h
	mov ds,ax
	mov al,ds:[0]
	mov bl,ds:[1]
	mov cl,ds:[2]
	mov dl,ds:[3]
	
	mov ax,4c00h
	int 21h
code ends
end

或者用类似如下的形式也可以mov al,[bx]，中括号里面为变量，而不是常量0，如果要用ss段寄存器，那么前面必须有ss:[bx]，而默认中括号前啥都没有就是ds寄存器

assume cs:code
code segment
	mov ax,2000h
	mov ds,ax
	mov bx,0
	mov al,[bx]
	mov al,ss:[bx]
	
	mov ax,4c00h
	int 21h
code ends
end

注意：我们以后写代码用200:2FF这段空间，这个空间是没有用的，向别的内存空间中写入可能直接会导致死机或者其他的事情。。。

作业:
将内存ffff:0到fff:c的单元中的数据复制到0:200到0:20c单元中，注意：0:200到0:20c等价于0020:0到0020:c，因为合成后物理地址都一样
assume cs:code
code segment
	mov bx,0
	mov cx,0CH
	
s:	mov ax,0ffffh
	mov ds,ax
	mov dx,[bx]
	
	mov ax,0020h
	mov ds,ax
	mov [bx],dx
	add bx,2
	loop s
	
	mov ax,4c00h
	int 21h
code ends
end
;除了上述方法，还可以用es这个段地址寄存器来实现
assume cs:code
code segment
	mov bx,0
	mov cx,0CH
	
	mov ax,0ffffh
	mov ds,ax
	
	mov ax,0020h
	mov es,ax
	
s:	mov dx,[bx]
	mov es:[bx],dx   ;es不可省略，分号是汇编的注释符号
	add bx,2
	loop s
	
	mov ax,4c00h
	int 21h
code ends
end

上面的程序中是只有一个段的code segment，也就是代码段，但是有的包含多个段，比如，同时包含代码段和数据段，那么数据段肯定要在内存中占用一定的空间，操作系统就会给数据段分配一定的空间，注意：操作系统给程序分配的空间都是安全的，因为操作系统不会让一个程序所用空间和其他程序或操作系统自己空间相冲突

程序获取所需要的空间有2种方法：

  1. 加载程序的时候为程序分配内存
  2. 在程序执行过程中向系统申请（c语言的malloc）

Start用法

assume cs:code
code segment
	dw 123h,0456h,0abch
	mov ax,0ffffh
	mov ds,ax 
code ends
end

编译，调试上面的程序，我们发现并不是从mov ax,0ffffh开始执行的，如果想从正常从mov ax,0ffffh开始执行的话，需要加上start

assume cs:code
code segment
	dw 123h,0456h,0abch ;定义字型数据，数据之间有逗号隔开
start:	mov ax,0ffffh  ;这里的start可以定义成任意名称
		mov ds,ax 
code ends
end start

重新编译，调试，就是从mov ax,0ffffh正常开始执行了，并且我们发现cs:ip变成了076A:0006，偏移地址ip变成了0006，恰好我们定义了三个字型数据，也就是6个字节

查看076a:0的地址内容，就发现了我们的字型数据

定义栈段

assume cs:code,ds:data,ss:stack
data segment
	dw 0123h,0456h
data ends

stack segment
	dw 0,0,0,0,0,0,0,0  ;栈的内存空间
stack ends

code segment
	start: 
	
		    mov ax,data
			mov ds,ax
			mov ax,stack
			mov ss,ax
			mov sp,10h
		    mov ax,0
		    mov ax,4c00h
		    int 21h
		
code ends
end start

    dw 0,0,0,0,0,0,0,0
